import chromadb
from sentence_transformers import SentenceTransformer
import torch
from transformers import AutoTokenizer, AutoModelForCausalLM
import textwrap
import time  # --- BENCHMARK ADDITION: For high-precision timers ---

# --- CONFIGURATION ---
DB_PATH = "./chroma_db_ai_papers_mpnet"
COLLECTION_NAME = "ai_ml_papers_mpnet_base_v2"
EMBEDDING_MODEL_NAME = 'sentence-transformers/all-mpnet-base-v2'
LLM_MODEL_ID = "llama"

print("Initializing the RAG query engine...")

device = 'cuda' if torch.cuda.is_available() else 'cpu'
print(f"Using device: {device}")

print(f"Loading vector database from: {DB_PATH}")
client = chromadb.PersistentClient(path=DB_PATH)
collection = client.get_collection(name=COLLECTION_NAME)

print(f"Loading embedding model: {EMBEDDING_MODEL_NAME}")
embedding_model = SentenceTransformer(EMBEDDING_MODEL_NAME, device=device)

print(f"Loading LLM for generation: {LLM_MODEL_ID}...")
tokenizer = AutoTokenizer.from_pretrained(LLM_MODEL_ID)
llm = AutoModelForCausalLM.from_pretrained(
    LLM_MODEL_ID,
    torch_dtype=torch.bfloat16,
    device_map="auto",
)
print("Initialization complete. You can now ask questions.")

# --- THE RAG FUNCTION WITH BENCHMARKING ---
def ask_question(query, max_new_tokens=512, num_sources=5):
    print("\n--- New Query ---")
    print(f"Query: {query}")

    # Reset max memory allocation statistics prior to query execution
    if torch.cuda.is_available():
        torch.cuda.reset_peak_memory_stats()

    start_total = time.time()  # --- BENCHMARK ADDITION ---

    # 1. Embed and Retrieve (Retrieval Latency)
    start_retrieval = time.time()  # --- BENCHMARK ADDITION ---
    
    query_embedding = embedding_model.encode(query, convert_to_tensor=True).tolist()
    
    print("Step 1: Retrieving relevant documents from the database...")
    results = collection.query(
        query_embeddings=[query_embedding],
        n_results=num_sources
    )
    
    end_retrieval = time.time()  # --- BENCHMARK ADDITION ---
    retrieval_latency_ms = (end_retrieval - start_retrieval) * 1000

    documents = results['documents'][0]
    metadatas = results['metadatas'][0]
    
    if not documents:
        return "I could not find any relevant information in the database.", [], {}

    # 2. Prompt Construction
    print("Step 2: Constructing prompt for the LLM...")
    context_str = ""
    for i, doc in enumerate(documents):
        context_str += f"Source {i+1} (from paper titled '{metadatas[i]['title']}'):\n"
        context_str += f"{doc}\n\n"
        
    prompt_template = f"""
You are an expert research assistant. Your task is to answer the user's question based ONLY on the following context from scientific papers.
Synthesize the information from the provided sources and give a clear, concise answer.
After your answer, list the titles of the papers you used as sources.

CONTEXT:
{context_str}

QUESTION:
{query}

ANSWER:
"""
    
    # 3. Text Generation (Generation Latency and Throughput)
    print("Step 3: Generating answer with the LLM...")
    messages = [{"role": "user", "content": prompt_template}]

    input_ids = tokenizer.apply_chat_template(
        messages,
        add_generation_prompt=True,
        return_tensors="pt"
    ).to(llm.device)
    
    start_generation = time.time()  # --- BENCHMARK ADDITION ---
    
    outputs = llm.generate(
        input_ids,
        max_new_tokens=max_new_tokens,
        do_sample=True,
        temperature=0.6,
        top_p=0.9,
    )
    
    end_generation = time.time()  # --- BENCHMARK ADDITION ---
    generation_latency_s = end_generation - start_generation
    end_total = time.time()  # --- BENCHMARK ADDITION ---

    # Token Count Calculation for Throughput Analysis
    response_tokens = outputs[0][input_ids.shape[-1]:]
    tokens_generated = len(response_tokens)
    throughput_tokens_per_s = tokens_generated / generation_latency_s if generation_latency_s > 0 else 0
    total_runtime_s = end_total - start_total

    # VRAM Memory Scaling Analysis
    peak_vram_gb = 0
    if torch.cuda.is_available():
        peak_vram_gb = torch.cuda.max_memory_allocated() / (1024 ** 3)

    answer = tokenizer.decode(response_tokens, skip_special_tokens=True)
    source_titles = sorted(list(set([meta['title'] for meta in metadatas])))
    
    # Pack metrics for reporting
    metrics = {
        "retrieval_latency_ms": retrieval_latency_ms,
        "generation_latency_s": generation_latency_s,
        "total_runtime_s": total_runtime_s,
        "tokens_generated": tokens_generated,
        "throughput_tokens_per_s": throughput_tokens_per_s,
        "peak_vram_gb": peak_vram_gb
    }
    
    return answer, source_titles, metrics
